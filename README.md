# Проект
Стартовый boot файл - main.py. Файл отработки программы и парсинга команд - engine.py. Файл со всеми константами - constants.py. core.py - файл с реализацией взаимодействия пользователя с базой данных. utils.py - файл с дополнительным функционалом.
Также есть функционал автоматических обработок ошибок и подтверждение рискованных действий с помощью декораторов, находящихся в файле decorators.py
# Работа с базой данных
Запуск режима: make project или project

help - отображение всех доступных комманд
exit - окончание выполнение программы
create_table <name> <column1name:type> <column2name:type> .. - создание таблицы данных
list_tables - отображение всех созданных таблиц
drop_table <name> - удаление таблицы данных по имени
# Работа с таблицами и данными
Запуск режима: database

create_table <имя_таблицы> <столбец1:тип> <столбец2:тип> - создать таблицу
insert into <имя_таблицы> values (<значение1>, <значение2>, ...) - создать запись
select from <имя_таблицы> where <столбец> = <значение> - прочитать записи по условию
select from <имя_таблицы> - прочитать все записи
update <имя_таблицы> set <столбец1> = <новое_значение1> where <столбец_условия> = <значение_условия> - обновить запись
delete from <имя_таблицы> where <столбец> = <значение> - удалить запись
info <имя_таблицы> - вывести информацию о таблице
help - справочная информация
exit - выйти из программы
